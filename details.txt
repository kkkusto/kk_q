import java.text.SimpleDateFormat;
import java.util.Date;

public class FilePathFormatter {

    public static String formatPath(String basePath) {
        // Get today's date
        Date today = new Date();
        SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMdd");
        String formattedDate = dateFormat.format(today);

        // Create the filename with today's date
        String fileName = formattedDate + ".GG.txt";

        // Combine base path and filename
        if (!basePath.endsWith("/")) {
            basePath += "/";
        }
        return basePath + fileName;
    }

    public static void main(String[] args) {
        // Example usage
        String path = "/your/path/here";
        String fullPath = formatPath(path);
        System.out.println("Formatted Path: " + fullPath);
    }
}
